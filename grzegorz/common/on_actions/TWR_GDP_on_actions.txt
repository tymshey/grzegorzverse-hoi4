on_actions = {
	on_startup = {
		effect = {
			every_country = {
				if = {
					limit = {
						NOT = { is_in_array = { global.GDP_list_array = THIS } }
					}
					add_to_array = { global.GDP_list_array = THIS }
				}
				calculate_GDP = no
			}
		}
	}
	on_monthly = {
		effect = {
			calculate_GDP = yes
			force_update_map_mode = { mapmode = TWR_government_type_map_mode }
			force_update_map_mode = { mapmode = TWR_industrialisation_map_mode }
		}
	}
}
